108 template<class T>
109 T* assign(EntityId id) {
110 int component_id = Component::get_id<T>();
111
112 // check if the entity is not deleted
113
114
115 if (component_pools.size() <= component_id) { // not enough component pool
116 component_pools.resize(component_pools.size() + 1, nullptr);
117 }
118 if (component_pools[component_id] == nullptr) {
119 component_pools[component_id] = new ComponentPool(sizeof(T));
120 }
121
122 T* pComponent = new (component_pools[component_id]->get(id)) T();
123
124
125 entities[id].mask.set(component_id);
126 return pComponent;
127 }
